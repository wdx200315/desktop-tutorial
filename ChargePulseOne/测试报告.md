```cpp
void IntegrationTest::testFullChargingFlow() {
    // 1. 车主扫码充电
    net.startCharge(chargerId, "auto", 0);
    wait(1000);
    
    // 2. 管理端看到充电桩状态变化
    // 预期: 状态从"online"变为"charging"
    auto status = getChargerStatus(chargerId);
    QVERIFY(status == "charging");
    
    // 3. 服务端实时推送数据
    // 预期: 订单数据每2秒更新一次
    int updates = 0;
    for (int i = 0; i < 5; i++) {
        wait(2000);
        if (hasNewData) updates++;
    }
    QVERIFY(updates >= 4);  // 至少4次更新
    
    // 4. 停止充电
    net.stopCharge(orderId);
    wait(1000);
    
    // 5. 验证订单状态
    auto order = getOrderDetail(orderId);
    QVERIFY(order.status == "completed");
    QVERIFY(order.energy > 0);
    QVERIFY(order.amount > 0);
}
```

```cpp
void PerformanceTest::testConcurrentCharging() {
    // 模拟10个充电桩同时工作
    const int CHARGER_COUNT = 10;
    const int DURATION_SECONDS = 60;
    
    // 启动充电
    for (int i = 1; i <= CHARGER_COUNT; i++) {
        startCharging(i);
    }
    
    // 监控1分钟
    auto start = std::chrono::steady_clock::now();
    while (elapsed < DURATION_SECONDS) {
        // 收集所有充电桩数据
        auto data = collectAllChargingData();
        processData(data);
        std::this_thread::sleep_for(std::chrono::seconds(1));
    }
    
    // 验证数据一致性
    QVERIFY(allDataConsistent == true);

}
```

```cpp
void PerformanceTest::testWebSocketPushPerformance() {
    // 模拟100个客户端连接
    const int CLIENT_COUNT = 100;
    
    // 启动WebSocket服务器
    wsServer.start(9011);
    
    // 连接100个客户端
    for (int i = 0; i < CLIENT_COUNT; i++) {
        connectClient(i);
    }
    
    // 广播消息

    auto start = std::chrono::high_resolution_clock::now();
```
